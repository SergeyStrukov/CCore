ToMemBase is a Partial class, designed to be a base class for "... to memory" classes. 


class ToMemBase : NoCopy
 {
   uint8 *ptr;
   ulen len;
   
  protected: 
  
   uint8 * alloc(ulen len); // one-time call
   
  public:
  
   ToMemBase();
   
   ~ToMemBase();
   
   const uint8 * getPtr() const { return ptr; }
 
   ulen getLen() const { return len; }
   
   // swap/move objects

   void objSwap(ToMemBase &obj);
   
   explicit ToMemBase(ToMoveCtor<ToMemBase> obj);
   
   ToMemBase * objMove(Place<void> place);
 };

It has the protected method alloc(), it must be called no more than once in a derived class. Default constructor initialize the class in the null state. Destructor releases the allocated memory. The class is swappable and movable.
