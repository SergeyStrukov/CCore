Function a lightweight callback. 

template <class R,class ... AA>
class Function<R (AA...)>
 {
  public: 
   
   // constructors

   Function(); // null function

   Function(NothingType); // null function

   Function(R (*func)(AA...)); // callback to static function

   Function(R (*func)(Handle ctx,AA...),Handle ctx); // callback to static function with handle

   template <class T>
   Function(T *obj_,R (T::*method_)(AA...)); // callback to object with method 

   // methods

   bool operator + () const; // not null
 
   bool operator ! () const; // is null

   template <class ... TT>
   R operator () (TT && ... tt); // do call
 };

You can bind Function to a static function, a static function with handle or an object with a method. Make sure the handle or the object is alive when you do call the function — there is no life-time control in Function. The object type must be derived from the Funchor base class. 

struct Funchor
 {
  template <class T,class R,class ... AA>
  static Function<R (AA...)> FunctionOf(T *obj,R (T::*method)(AA...)); 
 };

The static method FunctionOf of this class can be used to create a Function object.

There is a noncopyable base type Funchor_nocopy: 


typedef NoCopyBase<Funchor> Funchor_nocopy;

The following pattern to work with Functions is used in CCore: 

class C : public Funchor
 {
  public:

   void method();

   Function<void (void)> function_method() { return FunctionOf(this,&C::method); }
 };
