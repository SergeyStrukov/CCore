PrintAbort() is a simple helper function. It formats a message in a temporary buffer and then calls Abort(). 


template <class ... TT>
void PrintAbort(const char *format,const TT & ... tt)
 {
  char buf[TextBufLen];
  PrintBuf out(Range(buf));
  
  Printf(out,format,tt...);
  
  Abort(out.close());
 }
